# -*- coding: utf8 -*-\nimport requests\nfrom bs4 import BeautifulSoup\nimport json\nimport re\nimport datetime\n\n# 方糖推送（需要输入方糖推送的sckey）sckey = ''\nftqqUrl = "https://sc.ftqq.com/%s.send" % (sckey)\n\n# wxpusher推送（需要输入wxpusher的apptoken）appToken = ""\nwxurl = "http://wxpusher.zjiecode.com/api/send/message"\n\ngetAllCourseDataUrl = 'http://mooc1-api.chaoxing.com/mycourse/backclazzdata?view=json&rss=1'\n\nheaders = {\n    'Host': 'mooc1-api.chaoxing.com',\n    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',\n    'Cookie': '在这里输入你的学习通cookie',\n    'User-Agent': '在这里输入你的user-Agent',\n    'Accept-Language': 'zh-cn',\n    'Accept-Encoding': 'gzip, deflate, br',\n    'Connection': 'keep-alive'\n}\nwx_header = {\n    "Content-Type": "application/json",\n}\n\n\ndef checkHomework():\n    status = "未完成作业概览：\\n"\n    courses = getDataUrl()\n    allTodoWork = []\n    for name in courses:\n        url = courses[name]\n        homework = requests.get(url, headers=headers).text\n        soup = BeautifulSoup(homework, 'lxml')\n        work = soup.find_all('li')\n        for i in range(len(work)):\n            if work.div.span.string == "未交" and work.find('span', class_='fr') is not None:\n                # status = status + '\\n' + name + '\\n'\n                restTime = work.find('span', class_='fr').string\n                # 剩余具体小时数\n                restHour = re.findall(r"剩余(.+?)小时", restTime)\n                # 剩余多出来的分钟数\n                restMinute = re.findall(r"小时(.+?)分钟", restTime)\n                # 当前时间加上剩余小时\n                endTime1 = datetime.datetime.now() + datetime.timedelta(hours=int(restHour[0]))\n                # 最后加上分钟数即可算出具体的截止时间\n                endTime2 = endTime1 + datetime.timedelta(minutes=int(restMinute[0]))\n                # 当前作业状态\n                workStatus = '\\n' + name + '\\n' + work[\n                    i].div.p.string + '\\n' + restTime + '\\n' + '截止时间：' + endTime2.strftime(\n                    "%Y-%m-%d %H:%M:%S") + '\\n'\n                allTodoWork.append(workStatus)\n    # 重新排序，时间截止日期近的排在前面\n    allTodoWork.sort(key=lambda t: int(re.findall(r"剩余(.+?)小时", t)[0]))\n    for i in range(len(allTodoWork)):\n        status = status + allTodoWork\n    print(status)\n    # payload = {'text': '作业未完成通知', 'desp': status}\n    # requests.post(ftqqUrl, params=payload)\n    body = {\n        "appToken": appToken,\n        "content": status,\n        "summary": "",\n        "contentType": 1,\n        "topicIds": [],\n        "uids": ["在这里需要输入你的wxpusher的UID"],\n    }\n    requests.post(wxurl, headers=wx_header, data=json.dumps(body))\n\n\n# 获取所有本学期内的课程名及对应url\ndef getDataUrl():\n    allCourse = requests.get(getAllCourseDataUrl, headers=headers).text\n    allCourseJson = json.loads(allCourse)\n    channelList = allCourseJson['channelList']\n    cpi = ''\n    for c in range(len(channelList)):\n        # 获取最近添加的课程的cpi，使用该cpi来判断其它课程是否为本学期的\n        if 'cpi' in channelList[c]:\n            cpi = channelList[c]['cpi']\n            break\n    realCourse = []\n    finalCourseData = {}\n    for i in range(len(channelList)):\n        if 'cpi' in channelList and channelList['cpi'] == cpi:\n            realCourse.append(channelList['content']['course']['data'][0])\n\n    for t in range(len(realCourse)):\n        courseName = realCourse[t]['name']\n        courseSquareUrl = realCourse[t]['courseSquareUrl']\n        courseId = re.findall(r'courseId=(.+?)&', courseSquareUrl)[0]\n        classId = re.findall(r'classId=(.+?)&', courseSquareUrl)[0]\n        courseWorkUrl = 'https://mooc1-api.chaoxing.com/work/task-list?courseId=' + courseId + '&classId=' + classId + '&cpi=' + str(\n            cpi)\n        tmpDict = {courseName: courseWorkUrl}\n        finalCourseData.update(tmpDict)\n    return finalCourseData\n\n\ndef main_handler(event, context):\n    return checkHomework()\n\n\nif __name__ == '__main__':\n    checkHomework()\n
